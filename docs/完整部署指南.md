### 步骤 2：创建数据库表

#### 2.1 打开 SQL Editor
在 Supabase 仪表盘中，点击左侧菜单的 **SQL Editor**

#### 2.2 创建表结构
复制以下 SQL 并执行：

```sql
-- ============================================================
-- 留情局 - 核心数据表
-- ============================================================

-- 用户档案表
CREATE TABLE IF NOT EXISTS user_profiles (
    id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
    username TEXT UNIQUE NOT NULL,
    display_name TEXT,
    avatar_url TEXT,
    bio TEXT,
    points INTEGER DEFAULT 0,
    is_admin BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 论坛帖子表
CREATE TABLE IF NOT EXISTS forum_posts (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    author_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
    title TEXT NOT NULL,
    content TEXT NOT NULL,
    tags TEXT[],
    views INTEGER DEFAULT 0,
    likes INTEGER DEFAULT 0,
    reply_count INTEGER DEFAULT 0,
    is_pinned BOOLEAN DEFAULT FALSE,
    is_locked BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 帖子回复表
CREATE TABLE IF NOT EXISTS post_replies (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    post_id UUID REFERENCES forum_posts(id) ON DELETE CASCADE,
    author_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
    content TEXT NOT NULL,
    likes INTEGER DEFAULT 0,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 资源分享表
CREATE TABLE IF NOT EXISTS resources (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    author_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
    title TEXT NOT NULL,
    description TEXT,
    resource_type TEXT NOT NULL,
    url TEXT,
    file_path TEXT,
    tags TEXT[],
    download_count INTEGER DEFAULT 0,
    quality_score INTEGER DEFAULT 0,
    is_verified BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 通知表
CREATE TABLE IF NOT EXISTS notifications (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
    type TEXT NOT NULL,
    title TEXT NOT NULL,
    content TEXT,
    link TEXT,
    is_read BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 积分历史表
CREATE TABLE IF NOT EXISTS points_history (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
    action TEXT NOT NULL,
    points INTEGER NOT NULL,
    description TEXT,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 情报数据表（AI采集）
CREATE TABLE IF NOT EXISTS intelligence_posts (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    title TEXT NOT NULL,
    content TEXT,
    source_url TEXT,
    university TEXT,
    category TEXT,
    published_at TIMESTAMPTZ,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- ============================================================
-- 索引创建
-- ============================================================
CREATE INDEX IF NOT EXISTS idx_posts_author ON forum_posts(author_id);
CREATE INDEX IF NOT EXISTS idx_posts_created ON forum_posts(created_at DESC);
CREATE INDEX IF NOT EXISTS idx_replies_post ON post_replies(post_id);
CREATE INDEX IF NOT EXISTS idx_resources_author ON resources(author_id);
CREATE INDEX IF NOT EXISTS idx_notifications_user ON notifications(user_id, is_read);
CREATE INDEX IF NOT EXISTS idx_points_user ON points_history(user_id, created_at DESC);

-- ============================================================
-- Row Level Security (RLS) 配置
-- ============================================================

-- 启用 RLS
ALTER TABLE user_profiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE forum_posts ENABLE ROW LEVEL SECURITY;
ALTER TABLE post_replies ENABLE ROW LEVEL SECURITY;
ALTER TABLE resources ENABLE ROW LEVEL SECURITY;
ALTER TABLE notifications ENABLE ROW LEVEL SECURITY;
ALTER TABLE points_history ENABLE ROW LEVEL SECURITY;
ALTER TABLE intelligence_posts ENABLE ROW LEVEL SECURITY;

-- 用户档案：所有人可读取，用户可更新自己的
CREATE POLICY "Profiles are viewable by everyone" ON user_profiles FOR SELECT USING (true);
CREATE POLICY "Users can update own profile" ON user_profiles FOR UPDATE USING (auth.uid() = id);

-- 帖子：所有人可读取，认证用户可创建
CREATE POLICY "Posts are viewable by everyone" ON forum_posts FOR SELECT USING (true);
CREATE POLICY "Authenticated users can create posts" ON forum_posts FOR INSERT WITH CHECK (auth.role() = 'authenticated');
CREATE POLICY "Authors can update own posts" ON forum_posts FOR UPDATE USING (auth.uid() = author_id);

-- 回复：所有人可读取，认证用户可创建
CREATE POLICY "Replies are viewable by everyone" ON post_replies FOR SELECT USING (true);
CREATE POLICY "Authenticated users can create replies" ON post_replies FOR INSERT WITH CHECK (auth.role() = 'authenticated');
CREATE POLICY "Authors can update own replies" ON post_replies FOR UPDATE USING (auth.uid() = author_id);

-- 资源：所有人可读取，认证用户可创建
CREATE POLICY "Resources are viewable by everyone" ON resources FOR SELECT USING (true);
CREATE POLICY "Authenticated users can create resources" ON resources FOR INSERT WITH CHECK (auth.role() = 'authenticated');
CREATE POLICY "Authors can update own resources" ON resources FOR UPDATE USING (auth.uid() = author_id);

-- 通知：用户只能看到自己的通知
CREATE POLICY "Users can view own notifications" ON notifications FOR SELECT USING (auth.uid() = user_id);
CREATE POLICY "Users can update own notifications" ON notifications FOR UPDATE USING (auth.uid() = user_id);

-- 积分历史：用户只能看到自己的
CREATE POLICY "Users can view own points history" ON points_history FOR SELECT USING (auth.uid() = user_id);

-- 情报：所有人可读取
CREATE POLICY "Intelligence posts are viewable by everyone" ON intelligence_posts FOR SELECT USING (true);

-- ============================================================
-- 自动创建用户档案的 Trigger
-- ============================================================
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS trigger AS $$
BEGIN
    INSERT INTO public.user_profiles (id, username, display_name)
    VALUES (new.id, new.raw_user_meta_data->>'username', new.raw_user_meta_data->>'full_name');
    RETURN new;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE TRIGGER on_auth_user_created
    AFTER INSERT ON auth.users
    FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();
```

#### 2.3 执行 SQL
点击 "Run" 按钮执行

## 第三阶段：部署配置

### 方案 A：部署到 Netlify（推荐）

#### A1. 连接 GitHub 到 Netlify
1. 访问 https://app.netlify.com
2. 点击 "Add new site" → "Import an existing project"
3. 选择 "GitHub"
4. 授权 Netlify 访问您的 GitHub 账户
5. 选择 `liuqingju` 仓库

#### A2. 配置构建
在配置页面填写：

```
Build command: (留空)
Publish directory: ./
```

#### A3. 配置环境变量
1. 点击 "Environment variables"
2. 添加以下变量：

| Key | Value | 说明 |
|-----|-------|------|
| `SUPABASE_URL` | `https://xxxx.supabase.co` | 您的 Supabase 项目 URL |
| `SUPABASE_ANON_KEY` | `eyJhbGci...` | 您的 Supabase anon key |

#### A4. 部署
1. 点击 "Deploy site"
2. 等待构建完成（约1-2分钟）
3. Netlify 会为您生成一个随机子域名，如：`random-name.netlify.app`

#### A5. 配置域名（可选）
1. 在 Site settings → Domain management 中添加自定义域名
2. 按照提示配置 DNS

### 方案 C：部署到 GitHub Pages（免费）

#### C1. 启用 GitHub Pages
1. 在 GitHub 仓库页面，点击 **Settings**
2. 左侧菜单点击 **Pages**
3. 设置：
   - Source: Deploy from a branch
   - Branch: `main` (或 `master`)
   - Folder: `/ (root)`

#### C2. 等待部署
- 约 1-2 分钟后，网站将在 `https://yourusername.github.io/liuqingju` 可用

#### C3. 注意
- GitHub Pages 不支持服务器端功能
- 部分动态功能可能受限

## ❓ 常见问题

### Q1: 部署后页面样式错乱
**解决**：检查 CSS 文件路径，确保所有相对路径正确

### Q2: Supabase 连接失败
**解决**：
1. 检查环境变量是否正确配置
2. 确认 Supabase 项目是否正常运行
3. 检查 RLS 策略是否正确设置

### Q3: 用户无法注册/登录
**解决**：
1. 检查 Supabase Authentication 设置
2. 确认 Email 模板配置
3. 检查浏览器控制台错误信息

### Q4: 图片/资源无法加载
**解决**：
1. 检查资源 URL 是否正确
2. 确认 Supabase Storage 配置
3. 检查 CORS 设置
